### JavaScript（pages/publish/publish.js）

```javascript
Page({
  data: {
    form: {
      title: '',
      price: '',
      description: '',
      type: '普通',
      images: [],
      fromDorm: '',
      toDorm: ''
    },
    templates: [
      '全新数码配件，功能完好，价格实惠',
      '闲置书籍，保存完好，适合学习参考',
      '精美文创产品，独特设计，值得收藏'
    ],
    dorms: ['东区1栋', '东区2栋', '西区1栋', '西区2栋', '南区1栋', '南区2栋', '北区1栋', '北区2栋']
  },
  selectType(e) {
    const type = e.currentTarget.dataset.type;
    this.setData({ 'form.type': type });
  },
  bindInput(e) {
    const key = e.currentTarget.dataset.key;
    const value = e.detail.value;
    this.setData({ [`form.${key}`]: value });
  },
  chooseImage() {
    wx.chooseImage({
      count: 9 - this.data.form.images.length,
      sizeType: ['compressed'],
      sourceType: ['album', 'camera'],
      success: res => {
        const images = this.data.form.images.concat(res.tempFilePaths);
        this.setData({ 'form.images': images });
      }
    });
  },
  deleteImage(e) {
    const index = e.currentTarget.dataset.index;
    const images = this.data.form.images.filter((_, i) => i !== index);
    this.setData({ 'form.images': images });
  },
  useTemplate(e) {
    const index = e.currentTarget.dataset.index;
    const template = this.data.templates[index];
    this.setData({ 'form.description': template });
  },
  bindPickerChange(e) {
    const key = e.currentTarget.dataset.key;
    const value = this.data.dorms[e.detail.value];
    this.setData({ [`form.${key}`]: value });
  },
  async publish() {
    const { title, price, description, images, fromDorm, toDorm } = this.data.form;
    if (!title || !price || !description || images.length === 0 || !fromDorm || !toDorm) {
      wx.showToast({ title: '请填写完整信息', icon: 'none' });
      return;
    }
    wx.showLoading({ title: '发布中...' });
    try {
      const uploadTasks = images.map((path, index) => 
        wx.cloud.uploadFile({
          cloudPath: `boxes/${Date.now()}-${index}.png`,
          filePath: path
        })
      );
      const uploadResults = await Promise.all(uploadTasks);
      const imageUrls = uploadResults.map(res => res.fileID);
      await wx.cloud.callFunction({
        name: 'publishBox',
        data: { ...this.data.form, images: imageUrls }
      });
      wx.hideLoading();
      wx.showToast({ title: '发布成功，等待审核' });
      wx.navigateBack();
    } catch (error) {
      wx.hideLoading();
      wx.showToast({ title: '发布失败', icon: 'none' });
    }
  }
});
```

Page({
  data: {
    box: null,
    loading: true
  },
  onLoad(options) {
    const boxId = options.id;
    this.loadBoxDetail(boxId);
  },
  loadBoxDetail(boxId) {
    wx.showLoading({ title: '加载中...' });
    wx.cloud.callFunction({
      name: 'getBoxDetail',
      data: { boxId },
      success: res => {
        wx.hideLoading();
        this.setData({ box: res.result, loading: false });
      },
      fail: err => {
        wx.hideLoading();
        wx.showToast({ title: '加载失败', icon: 'none' });
        this.setData({ loading: false });
      }
    });
  },
  formatTime(timestamp) {
    const date = new Date(timestamp);
    return `${date.getFullYear()}-${(date.getMonth() + 1).toString().padStart(2, '0')}-${date.getDate().toString().padStart(2, '0')}`;
  },
  addToCollection() {
    wx.cloud.callFunction({
      name: 'addCollection',
      data: { boxId: this.data.box._id },
      success: res => {
        wx.showToast({ title: '收藏成功' });
      },
      fail: err => {
        wx.showToast({ title: '收藏失败', icon: 'none' });
      }
    });
  },
  buyBox() {
    const { box } = this.data;
    wx.cloud.callFunction({
      name: 'createOrder',
      data: {
        boxId: box._id,
        price: box.price,
        deliveryFee: 1
      },
      success: res => {
        wx.requestPayment({
          timeStamp: res.result.timeStamp,
          nonceStr: res.result.nonceStr,
          package: res.result.package,
          signType: res.result.signType,
          paySign: res.result.paySign,
          success: () => {
            wx.showToast({ title: '支付成功' });
            wx.navigateTo({ url: '/pages/orders/orders' });
          },
          fail: () => {
            wx.showToast({ title: '支付失败', icon: 'none' });
          }
        });
      },
      fail: err => {
        wx.showToast({ title: '创建订单失败', icon: 'none' });
      }
    });
  }
});
