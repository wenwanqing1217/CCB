## 1. 自动捐赠定时任务（triggerAutoDonate）

**文件路径：** `cloudfunctions/triggerAutoDonate/index.js`

```javascript
const cloud = require('wx-server-sdk')                     // 引入云开发SDK
cloud.init({ env: cloud.DYNAMIC_CURRENT_ENV })            // 初始化云开发环境
const db = cloud.database()                               // 获取数据库实例
const _ = db.command                                      // 获取数据库操作符

exports.main = async (event, context) => {                // 云函数入口函数
  try {
    const fifteenDaysAgo = Date.now() - 15 * 24 * 60 * 60 * 1000  // 计算15天前的时间戳
    const boxesToDonate = await db.collection('boxes')     // 查询需要自动捐赠的盲盒
      .where({ status: 'active', publish_time: _.lt(fifteenDaysAgo) })  // 条件：状态为active且发布时间超过15天
      .get()                                              // 执行查询
    for (const box of boxesToDonate.data) {               // 遍历需要捐赠的盲盒
      await db.collection('boxes')                        // 更新盲盒状态
        .doc(box._id)                                     // 根据ID更新
        .update({ data: { status: 'donated_pending' } })   // 状态改为donated_pending
      await db.collection('donations')                    // 添加到捐赠记录
        .add({ data: {
          box_id: box._id,                                // 盲盒ID
          donor_id: box._openid,                          // 捐赠者ID
          receiver_id: null,                              // 接收者ID（待分配）
          feedback_img: '',                               // 反馈图片
          feedback_text: '',                              // 反馈文字
          create_time: Date.now()                         // 创建时间
        } })
    }
    return { success: true, donatedCount: boxesToDonate.data.length }  // 返回成功结果
  } catch (error) {
    console.error('自动捐赠失败', error)                   // 打印错误信息
    return { success: false, error: error.message }        // 返回失败结果
  }
}
```

## 2. 分区抢单核心（grabOrder）

**文件路径：** `cloudfunctions/grabOrder/index.js`

```javascript
const cloud = require('wx-server-sdk')                     // 引入云开发SDK
cloud.init({ env: cloud.DYNAMIC_CURRENT_ENV })            // 初始化云开发环境
const db = cloud.database()                               // 获取数据库实例

exports.main = async (event, context) => {                // 云函数入口函数
  try {
    const { orderId } = event                              // 获取订单ID
    const openid = cloud.getWXContext().OPENID            // 获取当前用户的openid
    const order = await db.collection('orders').doc(orderId).get()  // 查询订单信息
    if (!order.data || order.data.status !== 'pending') {  // 检查订单是否存在且状态为待抢单
      return { success: false, message: '订单不存在或已被抢' }  // 返回失败信息
    }
    const user = await db.collection('users').where({ _openid: openid, role: 'rider' }).get()  // 检查用户是否为骑手
    if (user.data.length === 0) {                         // 不是骑手
      return { success: false, message: '您不是骑手，无法抢单' }  // 返回失败信息
    }
    await db.collection('orders').doc(orderId).update({ data: {  // 更新订单状态
      status: 'grabbed',                                  // 状态改为已抢单
      riderOpenid: openid,                                // 记录骑手openid
      grabbed_at: new Date()                              // 记录抢单时间
    } })
    return { success: true, message: '抢单成功' }          // 返回成功信息
  } catch (error) {
    console.error('抢单失败:', error)                      // 打印错误信息
    return { success: false, message: '抢单失败', error: error.toString() }  // 返回失败信息
  }
}
```

## 4. 即时配送实现（deliverOrder）

**文件路径：** `cloudfunctions/deliverOrder/index.js`

```javascript
const cloud = require('wx-server-sdk')                     // 引入云开发SDK
cloud.init({ env: cloud.DYNAMIC_CURRENT_ENV })            // 初始化云开发环境
const db = cloud.database()                               // 获取数据库实例

exports.main = async (event, context) => {                // 云函数入口函数
  try {
    const { orderId, status, location } = event            // 获取订单ID、状态和位置
    const openid = cloud.getWXContext().OPENID            // 获取当前用户的openid
    const order = await db.collection('orders').doc(orderId).get()  // 查询订单信息
    if (!order.data) {                                    // 检查订单是否存在
      return { success: false, message: '订单不存在' }    // 返回失败信息
    }
    const user = await db.collection('users').where({ _openid: openid, role: 'rider' }).get()  // 检查用户是否为骑手
    if (user.data.length === 0) {                         // 不是骑手
      return { success: false, message: '您不是骑手，无法配送' }  // 返回失败信息
    }
    const updateData = { status: status, update_time: new Date() }  // 准备更新数据
    if (location) { updateData.location = location }      // 如果有位置信息，添加到更新数据
    await db.collection('orders').doc(orderId).update({ data: updateData })  // 更新订单状态
    if (status === 'completed') {                         // 如果状态为已完成
      await db.collection('users').doc(user.data[0]._id).update({  // 更新骑手信息
        data: { completed_orders: _.inc(1) }              // 完成订单数+1
      })
    }
    return { success: true, message: '配送状态更新成功' }  // 返回成功信息
  } catch (error) {
    console.error('配送状态更新失败:', error)              // 打印错误信息
    return { success: false, message: '配送状态更新失败', error: error.toString() }  // 返回失败信息
  }
}
```

## 6. 二手交换模块（exchangeItem）

**文件路径：** `cloudfunctions/exchangeItem/index.js`

```javascript
const cloud=require('wx-server-sdk')
cloud.init({env:cloud.DYNAMIC_CURRENT_ENV})
const db=cloud.database(),_=db.command

exports.main=async(e,c)=>{
  try{
    const{action,data}=e,openid=cloud.getWXContext().OPENID
    switch(action){
      case'publish':
        const{t,d,i,type,em}=data
        await db.collection('exchange_items').add({
          data:{title:t,description:d,images:i,type,exchangeMode:em,
            status:'pending',publisher_id:openid,
            create_time:Date.now(),update_time:Date.now()
          }
        })
        return{success:true,message:'发布成功，等待审核'}
      case'match':
        const{itemId}=data
        const item=await db.collection('exchange_items').doc(itemId).get()
        if(!item.data)return{success:false,message:'物品不存在'}
        const matched=await db.collection('exchange_items')
          .where({_id:_.neq(itemId),status:'available',type:item.data.type})
          .limit(5).get()
        return{success:true,data:matched.data}
      case'request':
        const{targetItemId}=data
        await db.collection('exchange_requests').add({
          data:{item_id:data.itemId,target_item_id:targetItemId,
            requester_id:openid,status:'pending',create_time:Date.now()
          }
        })
        return{success:true,message:'交换请求已发送'}
      case'confirm':
        const request=await db.collection('exchange_requests').doc(data.requestId).get()
        if(!request.data)return{success:false,message:'请求不存在'}
        await db.collection('exchange_requests').doc(data.requestId).update({
          data:{status:'confirmed'}
        })
        await db.collection('exchange_items')
          .where({_id:_.in([request.data.item_id,request.data.target_item_id])})
          .update({data:{status:'exchanged'}})
        return{success:true,message:'交换确认成功'}
      default:return{success:false,message:'未知操作'}
    }
  }catch(err){
    console.error('操作失败:',err)
    return{success:false,message:'操作失败',error:err.toString()}
  }
}
```
